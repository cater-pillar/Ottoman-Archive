CREATE VIEW occupation_count AS

SELECT COUNT(`household`.`household_id`), 
			`occupation`.`name`,
            `occ_income`.`name`,
FROM 
	`household` 
INNER JOIN 
	`occ_income` ON `household`.`household_id` = `occ_income`.`household_id`
INNER JOIN `occupation` ON `occupation`.`occupation_id` = `occ_income`.`occupation_id`
INNER JOIN 'occ_group` ON 'occ_group`.'occ_group_id` = `occupation`.'occ_group_id`
GROUP BY
	`occupation`.`name`
